MessagesPlaceholder

from langchain_core.prompts import MessagesPlaceholder

chat_prompt = ChatPromptTemplate.from_messages([
    ("system", "You are a helpful exam tutor."),
    MessagesPlaceholder("history"),
    ("human", "{input}"),
])
# history = list of prior HumanMessage / AIMessage objects

Implication: System = stable rules. Placeholder = growing memory. Few-shot examples go in ("ai", ...) turns before the real task.

Chains vs Graphs

  • Chain — linear: A → B → C (one path)
  • Graph — branching: classify → chat OR study (multiple paths)
classify → chat or study → END

StateGraph Basics

Notebook: Sections 9–11

from typing import TypedDict, Annotated
from langgraph.graph import StateGraph, START, END
from langgraph.graph.message import add_messages

class AgentState(TypedDict):
    messages: Annotated[list, add_messages]
    intent: str
    context_chunks: str

Nodes & Edges

def classify_node(state: AgentState) -> dict:
    # ChatPromptTemplate + llm → "chat" or "study"
    ...

def chat_node(state: AgentState) -> dict:
    # zero-shot general Q&A
    ...

def study_node(state: AgentState) -> dict:
    # few-shot flashcards + CoT for worked problems
    ...

graph = StateGraph(AgentState)
graph.add_node("classify", classify_node)
graph.add_node("chat", chat_node)
graph.add_node("study", study_node)

Conditional Routing

Notebook: Sections 12–14

def route(state: AgentState) -> str:
    return state["intent"]  # "chat" or "study"

graph.add_edge(START, "classify")
graph.add_conditional_edges("classify", route, {
    "chat": "chat",
    "study": "study",
})
graph.add_edge("chat", END)
graph.add_edge("study", END)

agent = graph.compile()

Test the Agent
